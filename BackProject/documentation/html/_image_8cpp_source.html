155 void Image::blur()
156 {
157  int iterator = 0;
158  int end = 0;
159 
160  float red = 0.f;
161  float green = 0.f;
162  float blue = 0.f;
163 
164 
165  // The critical rows are the first one and last one because they have no previous or next one.
166  // They have to be calculated separatly
167 
168  // The first row of pixels
169 
170  // The firs pixel has no previous pixel.
171  //To prevent to use a conditional, the first pixel is calculated as follows
172  add_components(red, green, blue, pixels[iterator]);
173  add_components(red, green, blue, pixels[iterator + 1]);
174  add_components(red, green, blue, pixels[iterator + width]);
175  add_components(red, green, blue, pixels[iterator + width + 1]);
176 
177  pixels[iterator].rgb_components.red = red * 0.25f;
178  pixels[iterator].rgb_components.green = green * 0.25f;
179  pixels[iterator].rgb_components.blue = blue * 0.25f;
180 
181  ++iterator;
182 
183 
184  // The rest of the first row
185  end = width;
186  while (iterator < width)
187  {
188  red = 0.f;
189  green = 0.f;
190  blue = 0.f;
191 
192  add_components(red, green, blue, pixels[iterator]);
193  add_components(red, green, blue, pixels[iterator - 1]);
194  add_components(red, green, blue, pixels[iterator + 1]);
195  add_components(red, green, blue, pixels[iterator + width]);
196  add_components(red, green, blue, pixels[iterator + width + 1]);
197  add_components(red, green, blue, pixels[iterator + width - 1]);
198 
199  pixels[iterator].rgb_components.red = red * 0.16f;
200  pixels[iterator].rgb_components.green = green * 0.16f;
201  pixels[iterator].rgb_components.blue = blue * 0.16f;
202 
203  ++iterator;
204  }
205 
206  end = width * height - width;
207 
208  // The body of the image
209  while (iterator < end)
210  {
211  red = 0.f;
212  green = 0.f;
213  blue = 0.f;
214 
215  add_components(red, green, blue, pixels[iterator]);
216  add_components(red, green, blue, pixels[iterator - 1]);
217  add_components(red, green, blue, pixels[iterator + 1]);
218  add_components(red, green, blue, pixels[iterator + width]);
219  add_components(red, green, blue, pixels[iterator + width + 1]);
220  add_components(red, green, blue, pixels[iterator + width - 1]);
221  add_components(red, green, blue, pixels[iterator - width]);
222  add_components(red, green, blue, pixels[iterator - width + 1]);
223  add_components(red, green, blue, pixels[iterator - width - 1]);
224 
225  pixels[iterator].rgb_components.red = red * 0.11f;
226  pixels[iterator].rgb_components.green = green * 0.11f;
227  pixels[iterator].rgb_components.blue = blue * 0.11f;
228 
229  ++iterator;
230  }
231 
232  // The last row of pixels
233  end += width;
234  while (iterator < end - 1)
235  {
236  red = 0.f;
237  green = 0.f;
238  blue = 0.f;
239 
240  add_components(red, green, blue, pixels[iterator]);
241  add_components(red, green, blue, pixels[iterator - 1]);
242  add_components(red, green, blue, pixels[iterator + 1]);
243  add_components(red, green, blue, pixels[iterator - width]);
244  add_components(red, green, blue, pixels[iterator - width - 1]);
245  add_components(red, green, blue, pixels[iterator - width + 1]);
246 
247  pixels[iterator].rgb_components.red = red * 0.16f;
248  pixels[iterator].rgb_components.green = green * 0.16f;
249  pixels[iterator].rgb_components.blue = blue * 0.16f;
250 
251  ++iterator;
252  }
253 
254  // The last pixel
255  red = 0.f;
256  green = 0.f;
257  blue = 0.f;
258 
259  add_components(red, green, blue, pixels[iterator]);
260  add_components(red, green, blue, pixels[iterator - 1]);
261  add_components(red, green, blue, pixels[iterator - width]);
262  add_components(red, green, blue, pixels[iterator - width - 1]);
263 
264  pixels[iterator].rgb_components.red = red * 0.25f;
265  pixels[iterator].rgb_components.green = green * 0.25f;
266  pixels[iterator].rgb_components.blue = blue * 0.25f;
267 
268 }

277 void Image::sobel_colour()
278 {
279  int end = width * height;
280  int iterator = 0;
281 
282  // set the start in the second row and the end in the previous of the last row
283  // This avoid extra conditional operations and the lost of information is not relevant
284  iterator += width + 1;
285  end -= (width + 1);
286 
287  std::vector<float> buffer(width * height);
288 
289  float min = 1.f;
290  float max = 0.f;
291 
292 
293  while (iterator < end)
294  {
295  float horizontal_edge = colour_difference(pixels[iterator - 1], pixels[iterator + 1]);
296  float diagonal_1_edge = colour_difference(pixels[iterator - width - 1], pixels[iterator + width + 1]);
297  float diagonal_2_edge = colour_difference(pixels[iterator - width + 1], pixels[iterator + width - 1]);
298  float vertical_edge = colour_difference(pixels[iterator - width], pixels[iterator + width]);
299 
300 
301  float val = horizontal_edge + diagonal_1_edge + diagonal_2_edge + vertical_edge;
302  if (val != 0)
303  {
304  int i = 0;
305  }
306  buffer[iterator] = val;
307 
308  max = val > max ? val : max;
309  min = val < min ? val : min;
310 
311 
312  ++iterator;
313  }
314 
315  iterator = 0;
316 
317  while (iterator < buffer.size())
318  {
319  buffer[iterator] = ((buffer[iterator] - min) / max - min);
320 
321  pixels[iterator].rgb_components.red = 1.f - buffer[iterator];
322  pixels[iterator].rgb_components.green = 1.f - buffer[iterator];
323  pixels[iterator].rgb_components.blue = 1.f - buffer[iterator];
324 
325  ++iterator;
326  }
327 }

406 float Image::compare(Image& other)
407 {
408  float index = 0.f;
409  float iterator = 0;
410  float max_count = width * height;
411 
412  float other_max = 0.0f;
413 
414  while (iterator < max_count)
415  {
416  other_max += other.get_pixel(iterator).rgb_components.red;
417 
418  index += abs(this->get_pixel(iterator).rgb_components.red - other.get_pixel(iterator).rgb_components.red);
419  ++iterator;
420  }
421 
422  return 100 * abs( 1 - (index / other_max));
423 }
