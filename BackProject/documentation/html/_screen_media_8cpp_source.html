32 void ScreenMedia::capture()
33 {
34  int x = 0;
35  int y = 0;
36  int width = 0;
37  int height = 0;
38 
39  RECT r;
40 
41  if (current_window)
42  {
43  GetClientRect(current_window, &r);
44  }
45  else
46  {
47  current_window = GetDesktopWindow();
48  GetWindowRect(current_window, &r);
49  }
50 
51  x += r.left;
52  y += r.top;
53  width = r.right - r.left;
54  height = r.bottom - r.top;
55 
56  // Device context where copy screen data
57  HDC display_dc = GetDC(nullptr);
58  HDC bitmap_dc = CreateCompatibleDC(display_dc);
59  HBITMAP bitmap = CreateCompatibleBitmap(display_dc, width, height);
60  HGDIOBJ null_bitmap = SelectObject(bitmap_dc, bitmap);
61 
62  // Copy the data from window to bitmap_dc
63  HDC window_dc = GetDC(current_window);
64  BitBlt(bitmap_dc, 0, 0, width, height, window_dc, x, y, SRCCOPY | CAPTUREBLT);
65 
66  // Clean
67  ReleaseDC(current_window, window_dc);
68  SelectObject(bitmap_dc, null_bitmap);
69  DeleteDC(bitmap_dc);
70 
71  render_pixmap = QtWin::fromHBITMAP(bitmap);
72 
73  DeleteObject(bitmap);
74  ReleaseDC(nullptr, display_dc);
75 }

102 void ScreenMedia::update_available_windows()
103 {
104  available_windows.clear();
105 
106  for (
107  HWND hwnd = GetTopWindow(NULL);
108  hwnd != NULL;
109  hwnd = GetNextWindow(hwnd, GW_HWNDNEXT)
110  )
111  {
112  if (!IsWindowVisible(hwnd))
113  {
114  continue;
115  }
116 
117  int length = GetWindowTextLength(hwnd);
118  if (length == 0)
119  continue;
120 
121  LPWSTR title = new TCHAR[length + 1];
122  GetWindowText(hwnd, title, length + 1);
123 
124  std::string name;
125  name.reserve(wcslen(title));
126 
127  while (*title)
128  {
129  name += (char)*title;
130  ++title;
131  }
132 
133  available_windows[name] = hwnd;
134 
135  //delete[] title;
136  }
137 }
