32 void Pixel::convert_xyz_to_luv(float* buffer)
33 {
34  float Xr = 0.33f;
35  float Yr = 0.33f;
36  float Zr = 0.33f;
37  float yr;
38  float us;
39  float vs;
40  float usr;
41  float vsr;
42  float e;
43  float k;
44 
45  yr = buffer[1] * 3.f; //< It's the same that divide buffer[1] / Yr
46  us = 4 * buffer[0] / (buffer[0] + 15 * buffer[1] + 3 * buffer[2]);
47  vs = 9 * buffer[1] / (buffer[0] + 15 * buffer[1] + 3 * buffer[2]);
48  usr = 4 * Xr / (Xr + 15 * Yr + 2 * Zr);
49  vsr = 9 * Yr / (Xr + 15 * Yr + 3 * Zr);
50 
51  e = 216.0f / 24389.f;
52  k = 24389.0f / 27.f;
53 
54  luv_components.l = yr > e ?
55  116.f * pow(yr, 0.33) - 16
56  : k * yr;
57 
58  luv_components.u = 13.0f * luv_components.l * (us - usr);
59  luv_components.v = 13.0f * luv_components.l * (vs - vsr);
60 
61 }

63 void Pixel::convert_luv_to_rgb()
64 {
65  float Xr = 0.33f;
66  float Yr = 0.33f;
67  float Zr = 0.33f;
68  float usr = 4 * Xr / (Xr + 15 * Yr + 2 * Zr);
69  float vsr = 9 * Yr / (Xr + 15 * Yr + 3 * Zr);
70 
71  float u_ = luv_components.u / (13 * luv_components.l) + usr;
72  float v_ = luv_components.v / (13 * luv_components.l) + vsr;
73  float Y = pow(((luv_components.l + 16) / 116), 3);
74  float X = (-9 * Y * u_) / (u_ - 4) * v_ - u_ * v_;
75  float Z = (9 * Y - 15 * v_ * Y - v_ * X) / 3 * v_;
76 
77  float r = (3.063218f) * X + (-1.393325f) * Y + (-0.475802f) * Z;
78  float g = (-0.969243f) * X + (1.875966f) * Y + (0.041555f) * Z;
79  float b = (0.067871f) * X + (-0.228834f) * Y + (1.069251f) * Z;
80 
81  rgb_components.red = r < 0.f ? 0.f : r > 1.f ? 1.f : r;
82  rgb_components.green = g < 0.f ? 0.f : g > 1.f ? 1.f : g;
83  rgb_components.blue = b < 0.f ? 0.f : b > 1.f ? 1.f : b;
84 }

325 void Pixel::convert_xyz_to_xyzs(float* buffer, float* simulation_modifiers)
326 {
327 
328  float original_0 = buffer[0];
329  float original_1 = buffer[1];
330  float original_2 = buffer[2];
331 
332  /*
333  The Color Blind Simulation function is
334  copyright (c) 2000-2001 by Matthew Wickline and the
335  Human-Computer Interaction Resource Network ( http://hcirn.com/ ).
336  It is used with the permission of Matthew Wickline and HCIRN,
337  and is freely available for non-commercial use. For commercial use, please
338  contact the Human-Computer Interaction Resource Network ( http://hcirn.com/ ).
339  */
340 
341  float cpu = simulation_modifiers[0];
342  float cpv = simulation_modifiers[1];
343  float am = simulation_modifiers[2];
344  float ayi = simulation_modifiers[3];
345 
346  float wx = 0.312713;
347  float wy = 0.329016;
348  float wz = 0.358271;
349 
350  float sum_xyz = original_0 + original_1 + original_2;
351  float cu = 0.f;
352  float cv = 0.f;
353 
354  if (sum_xyz != 0)
355  {
356  cu = original_0 / sum_xyz;
357  cv = original_1 / sum_xyz;
358  }
359 
360  float nx = wx * original_1 / wy;
361  float nz = wz * original_1 / wy;
362  float clm;
363 
364  if (cu < cpu)
365  {
366  clm = (cpv - cv) / (cpu - cu);
367  }
368  else
369  {
370  clm = (cv - cpv) / (cu - cpu);
371  }
372 
373  float clyi = cv - cu * clm;
374  float du = (ayi - clyi) / (clm - am);
375  float dv = (clm * du) + clyi;
376 
377  buffer[0] = du * original_1 / dv;
378  buffer[1] = original_1;
379  buffer[2] = (1 - (du + dv)) * original_1 / dv;
380 
381 }
